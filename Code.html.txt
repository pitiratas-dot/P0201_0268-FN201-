        function startCountdown() {
            const targetDate = new Date('May 12, 2026 09:00:00').getTime();
            const display = document.getElementById('sidebar-countdown');
            const statDays = document.getElementById('stat-days-left');

            setInterval(() => {
                const now = new Date().getTime();
                const distance = targetDate - now;

                if (distance < 0) {
                    display.innerText = "EXAM TIME!";
                    statDays.innerText = "0";
                    return;
                }

                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

                display.innerText = `${days}d ${hours}h ${minutes}m`;
                statDays.innerText = days;

                // Color logic
                display.classList.remove('text-success', 'text-warning', 'text-danger');
                if (days > 30) display.classList.add('text-success');
                else if (days > 7) display.classList.add('text-warning');
                else display.classList.add('text-danger');
            }, 1000);
        }

        function renderSidebarTopics() {
            const container = document.getElementById('topic-list-container');
            container.innerHTML = '';
            let masteredCount = 0;

            TOPICS.forEach(topic => {
                const progress = appState.topicsProgress[topic.id] || 0;
                if(progress === 100) masteredCount++;
                
                let badgeClass = 'bg-borderDark text-textMuted';
                let badgeDot = '🔴';
                if(progress > 0 && progress < 100) { badgeClass = 'bg-warning/20 text-warning'; badgeDot = '🟡'; }
                if(progress === 100) { badgeClass = 'bg-success/20 text-success'; badgeDot = '🟢'; }

                const el = document.createElement('div');
                el.className = 'group';
                el.innerHTML = `
                    <div class="flex justify-between items-center mb-1 text-xs">
                        <span class="text-textMain font-medium truncate pr-2" title="${topic.id}: ${topic.title}">${topic.id}: ${topic.title}</span>
                        <span class="${badgeClass} px-1.5 py-0.5 rounded text-[10px] font-bold whitespace-nowrap">${badgeDot} ${progress}%</span>
                    </div>
                    <div class="flex items-center gap-2">
                        <div class="h-1.5 w-full bg-borderDark rounded-full overflow-hidden">
                            <div class="h-full ${progress===100?'bg-success':'bg-accent'} transition-all duration-500" style="width: ${progress}%"></div>
                        </div>
                        <button class="text-[10px] bg-borderDark hover:bg-opacity-80 text-textMuted px-1.5 rounded transition-opacity opacity-0 group-hover:opacity-100" onclick="incrementTopic('${topic.id}')">+10</button>
                    </div>
                `;
                container.appendChild(el);

        }

        window.incrementTopic = function(id) {
            let val = appState.topicsProgress[id] || 0;
            if(val < 100) {
                appState.topicsProgress[id] = Math.min(100, val + 10);
                saveState();
                renderSidebarTopics();
            }
        };

        function updateOverallProgress() {
            const vals = Object.values(appState.topicsProgress);
            const sum = vals.reduce((a, b) => a + b, 0);
            const avg = vals.length ? Math.round(sum / vals.length) : 0;
            
            document.getElementById('overall-progress-text').innerText = `${avg}%`;
            
            const circle = document.getElementById('overall-progress-ring');
            const circumference = 2 * Math.PI * 40; // r=40
            const offset = circumference - (avg / 100) * circumference;
            circle.style.strokeDashoffset = offset;
        }

        function formatTime(seconds) {
            const m = Math.floor(seconds / 60).toString().padStart(2, '0');
            const s = (seconds % 60).toString().padStart(2, '0');
            return `${m}:${s}`;
        }

        function updatePomoUI() {
            pomoDisplay.innerText = formatTime(pomoTimeLeft);
            const circumference = 2 * Math.PI * 46; // r=46
            const percent = (pomoTotalTime - pomoTimeLeft) / pomoTotalTime;
            pomoRing.style.strokeDashoffset = circumference * percent;
            
            // Color based on mode
            if(pomoCurrentMode === 'study') pomoRing.setAttribute('stroke', '#22d3ee'); // accent
            else if(pomoCurrentMode === 'short') pomoRing.setAttribute('stroke', '#4ade80'); // success
            else pomoRing.setAttribute('stroke', '#fb923c'); // warning
        }

        function setPomoMode(minutes, mode) {
            clearInterval(pomoInterval);
            pomoIsRunning = false;
            pomoCurrentMode = mode;
            pomoTotalTime = minutes * 60;
            pomoTimeLeft = pomoTotalTime;
            pomoStartBtn.innerHTML = '<i data-lucide="play" class="w-5 h-5"></i> Start';
            lucide.createIcons({root: pomoStartBtn});
            
            pomoModeBtns.forEach(b => {
                if(b.dataset.mode === mode) {
                    b.classList.add('active', 'bg-borderDark', 'text-textMain', 'shadow-sm');
                    b.classList.remove('text-textMuted');
                } else {
                    b.classList.remove('active', 'bg-borderDark', 'text-textMain', 'shadow-sm');
                    b.classList.add('text-textMuted');
                }
            });
            updatePomoUI();
        }

                // Pause
                clearInterval(pomoInterval);
                pomoIsRunning = false;
                pomoStartBtn.innerHTML = '<i data-lucide="play" class="w-5 h-5"></i> Resume';
            } else {
                // Start
                pomoIsRunning = true;
                pomoStartBtn.innerHTML = '<i data-lucide="pause" class="w-5 h-5"></i> Pause';

                    sendDiscordNotification(`🍅 Study session started: **${taskName}**`, 'pomodoro');

                pomoInterval = setInterval(() => {
                    pomoTimeLeft--;
                    updatePomoUI();
                    
                    if(pomoTimeLeft <= 0) {
                        clearInterval(pomoInterval);
                        pomoIsRunning = false;
                        pomoStartBtn.innerHTML = '<i data-lucide="play" class="w-5 h-5"></i> Start';

                        if(pomoCurrentMode === 'study') {
                            appState.pomodorosToday++;

        window.copyText = function(text) {
            // Using modern clipboard API with fallback
            if(navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(text);
            } else {
                // Fallback
                let textArea = document.createElement("textarea");
                textArea.value = text;
                textArea.style.position = "fixed";
                textArea.style.left = "-999999px";
                textArea.style.top = "-999999px";
                document.body.appendChild(textArea);
                textArea.focus();
                textArea.select();
                try {
                    document.execCommand('copy');
                } catch (err) {
                    console.error('Fallback: Oops, unable to copy', err);
                }
                textArea.remove();
            }
            showToast('Formula copied to clipboard!', 'success');
        };

        // --- DISCORD INTEGRATION ---
        async function sendDiscordNotification(message, type = 'info') {
            const webhookUrl = appState.webhook;
            if (!webhookUrl) return; // Silent fail if not configured

            let color = 2282478; // info blue/cyan
            if (type === 'success') color = 4906880; // green
            if (type === 'warning') color = 16486972; // orange
            if (type === 'pomodoro') color = 15680580; // red

            const payload = {
                username: "FN201 Command Bot",
                avatar_url: "https://cdn-icons-png.flaticon.com/512/3135/3135692.png",
                embeds: [{
                    title: "📊 FN201 Study Update",
                    description: message,
                    color: color,
                    footer: { text: "FN201 Command Center · " + new Date().toLocaleTimeString() }
                }]
            };

            try {
                const res = await fetch(webhookUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                if (res.ok) {
                    logDiscordEvent(message);
                    return true;
                } else {
                    throw new Error('Webhook rejected');
                }
            } catch (err) {
                console.error('Discord Webhook Error:', err);
                showToast('Failed to send Discord notification. Check URL/CORS.', 'error');
                return false;
            }
        }

        function logDiscordEvent(msg) {
            const time = new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
            appState.discordLogs.unshift(`[${time}] ${msg}`);
            if(appState.discordLogs.length > 5) appState.discordLogs.pop();
            saveState();
            renderDiscordLogs();
        }

        function renderDiscordLogs() {

                container.innerHTML = '<p class="text-xs text-textMuted italic text-center mt-4">No recent messages sent.</p>';
                return;
            }
            container.innerHTML = appState.discordLogs.map(log => `<div class="text-xs text-textMuted bg-bgDark p-2 rounded border border-borderDark">${log}</div>`).join('');
        }

                    document.querySelectorAll('.tab-pane').forEach(p => p.classList.add('hidden'));
                    
                    // Activate clicked
                    const targetId = e.currentTarget.dataset.target;
                    e.currentTarget.classList.add('active');
                    document.getElementById(targetId).classList.remove('hidden');
                });
            });

            // Onboarding Setup
            document.getElementById('btn-start').addEventListener('click', () => {
                const name = document.getElementById('setup-name').value.trim();
                const webhook = document.getElementById('setup-webhook').value.trim();
                
                if(name) {
                    appState.name = name;
                    appState.onboarded = true;
                    if(webhook) appState.webhook = webhook;
                    
                    saveState();
                    updateGreeting();
                    
                    const modal = document.getElementById('welcome-modal');
                    const card = document.getElementById('welcome-card');
                    card.classList.add('scale-95');
                    modal.classList.add('opacity-0');
                    setTimeout(() => modal.classList.add('hidden'), 300);
                    
                    showToast('Welcome aboard!', 'success');
                    
                    // Sync sidebar webhook
                    document.getElementById('sidebar-webhook').value = webhook;
                } else {
                    showToast('Please enter your name.', 'warning');
                }
            });

            // Sidebar Webhook Save & Test
            document.getElementById('btn-test-discord').addEventListener('click', async () => {
                const url = document.getElementById('sidebar-webhook').value.trim();
                if(!url) { showToast('Enter a webhook URL first.', 'warning'); return; }
                
                appState.webhook = url;
                saveState();
                const btn = document.getElementById('btn-test-discord');
                btn.innerHTML = '<i data-lucide="loader" class="w-4 h-4 animate-spin"></i>';
                lucide.createIcons({root: btn.parentElement});
                
                const success = await sendDiscordNotification('🔔 Connection test successful! FN201 Command Center is linked.', 'info');
                if(success) showToast('Discord link successful!', 'success');
                
                btn.innerHTML = '<i data-lucide="bell-ring" class="w-4 h-4"></i>';
                lucide.createIcons({root: btn.parentElement});
            });

            // Notes Save
            const noteSelect = document.getElementById('note-topic-select');
            const noteEditor = document.getElementById('note-editor');
            
            noteSelect.addEventListener('change', () => {
                const topic = noteSelect.value;
                noteEditor.value = appState.notes[topic] || '';
            });

            document.getElementById('btn-save-note').addEventListener('click', () => {
                const topic = noteSelect.value;
                appState.notes[topic] = noteEditor.value;
                saveState();
                showToast(`Notes saved for ${topic}`, 'success');
            });

            // Discord Quick Messages
            document.querySelectorAll('.quick-msg-btn').forEach(btn => {
                btn.addEventListener('click', () => {
                    const msg = btn.innerText.trim();
                    sendDiscordNotification(msg, 'info');
                    showToast('Message sent to Discord', 'success');
                });
            });

            // Discord Custom Message
            document.getElementById('btn-send-custom-discord').addEventListener('click', () => {
                const input = document.getElementById('discord-custom-msg');
                const msg = input.value.trim();
                if(msg) {
                    sendDiscordNotification(`📝 Note: ${msg}`, 'info');
                    input.value = '';
                    showToast('Custom message sent', 'success');
                }
            });

            // Sync Checklist to Discord
            document.getElementById('btn-sync-discord').addEventListener('click', () => {
                const doneTasks = appState.checklist.filter(t => t.done);
                if(doneTasks.length === 0) {
                    showToast('No completed tasks to sync.', 'warning');
                    return;
                }
                const msgList = doneTasks.map(t => `~ ${t.text}`).join('\n');
                sendDiscordNotification(`**Progress Update!** Completed tasks:\n${msgList}`, 'success');
                showToast('Progress synced to Discord', 'success');
            });
        }

        // --- BOOTSTRAP ---
        document.addEventListener('DOMContentLoaded', init);

    </script>
